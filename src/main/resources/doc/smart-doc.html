                2.1.3. 用户注册

                URL: http://localhost:8081/user/register

                Type: POST

                Author: zhou

                Content-Type: application/json

                Description: 用户注册

                Body-parameters:

                ParameterTypeRequiredDescription

                account

                string

                true

                用户账号 Validate[max: 16; ]

                password

                string

                true

                用户密码 Validate[max: 20; regexp: ^(?=.*[A-Z])(?=.*\\d)\\S*$; ]

                sex

                string

                true

                用户性别

                mail

                string

                true

                用户邮箱

                nickName

                string

                true

                用户昵称 Validate[max: 20; regexp: ^[^\\s!@#$%^&*()_+{}\\[\\]:;<>,.?~\\\\/-]+$; ]

                Request-body:

                {
  "account": "",
  "password": "",
  "sex": "",
  "mail": "",
  "nickName": ""
}

                Response-example:

                Return void.

                Curl-example:

                curl -X POST -H 'Content-Type: application/json' -i http://localhost:8081/user/register --data '{
  "account": "",
  "password": "",
  "sex": "",
  "mail": "",
  "nickName": ""
}'
